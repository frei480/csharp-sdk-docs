# Поиск объектов  
  
---  
  
Поиск нужного объекта в документе осуществляется путем перебора всех существующих однотипных объектов. Для перебора объектов модели разных типов существует набор специализированных функций, возвращающих специальный контейнер, [ObjectContainer](T_TFlex_Model_ObjectContainer.md). Ниже приведен пример кода функции, в которой для всех линий изображения документа устанавливается красный цвет.
    
    
    private void MakeAllRed(Document document)
    {
        //Открываем блок изменения документа
      document.BeginChanges("Изменение цвета линий изображения");
      //Перебор всех линий изображения в документе
      foreach (Outline outline in document.GetOutlines())
      {
          //Устанавливаем красный цвет
          outline.Color = System.Convert.ToDouble(TFlex.Drawing.Color.Red);
      }
      //Закрываем блок изменения документа
      document.EndChanges();
    }
    
    
    Private Sub MakeAllRed(ByVal document As Document)
      'Открываем блок изменения документа
      document.BeginChanges("Изменение цвета линий изображения")
        'Перебор всех линий изображения в документе
        For Each outline As Outline In document.GetOutlines()
          'Устанавливаем красный цвет
          outline.Color = System.Convert.ToDouble(TFlex.Drawing.Color.Red)
        Next
      document.EndChanges()
    End Sub

В следующем примере выбираются грани всех твердых тел в сцене.

Примечание  
---  
Обратите внимание, что для перебора граней и других твердых тел, входящих в состав исходного твердого тела, используются геометрические данные объекта модели.  
      
    
    private void GetAllModelFaces(Document document)
    {
      //Создаем список .NET, в который будем добавлять выбранные грани
      List<ModelFace> facesList = new List<ModelFace>();
      //Перебор всех тел в сцене
      foreach (TFlex.Model.Model3D.BodyPart body in document.Bodies)
      {
        //Перебор всех твердых тел в сцене
        foreach (ModelBody Body in body.TopOperation.Geometry.Solid)
        {
          //Перебор всех граней твердых тел
          foreach (ModelFace face in Body.Faces)
          {
            //Добавляем грани в список
            facesList.Add(face);
          }
        }
      }
    }
    
    
    Private Sub GetAllModelFaces(ByVal document As Document)
      'Cоздаем список .NET, в который будем добавлять выбранные грани
      Dim facesList As New List(Of ModelFace)
      'Перебор всех тел в сцене
      For Each body As TFlex.Model.Model3D.BodyPart In document.Bodies
        'Перебор всех твердых тел в сцене
        For Each Body As ModelBody In body.TopOperation.Geometry.Solid
          'Перебор всех граней твердых тел
          For Each face As ModelFace In Body.Faces
            'Добавляем грани в список
            facesList.Add(face)
          Next
        Next
      Next
    End Sub
