# Cоздание линий изображения разного типа  
```csharp
public static void Outlines()//создание линий изображения разного типа
{
	//Получение активного документа
	Document document = TFlex.Application.ActiveDocument;
	//Открытие блока изменений документа
	document.BeginChanges("Построение разных линий изображения");

	//CircleOutline
	FreeNode fnCenter = new FreeNode(document, 200, 370);
	int r1 = 50;
	CircleOutline co = new  CircleOutline(document, fnCenter, r1);
	//CenterAxisOutline
	CenterAxisOutline cao = new CenterAxisOutline(document, co, false);
	//ConstructionOutline
	FreeNode fn1 = new FreeNode(document, 400, 320);            
	FreeNode fn2 = new FreeNode(document, 400, 420);
	ConstructionOutline conOutline = new ConstructionOutline(document, fn1, fn2);

	//AxisOutline
	AxisOutline axOutline = new AxisOutline(document, cao, conOutline);

	//ThreePointArcOutline - Линия изображения - дуга окружности, проходящая через три узла
	FreeNode fn3 = new FreeNode(document, 450, 320);            
	FreeNode fn4 = new FreeNode(document, 470, 340);            
	FreeNode fn5 = new FreeNode(document, 490, 320);            

	ThreePointArcOutline threePointArcOutline = new ThreePointArcOutline(document, fn3, fn4, fn5);

	//TwoPointArcOutline - Линия изображения - дуга окружности, проходящая через два узла с заданным радиусом
	FreeNode fn6 = new FreeNode(document, 450, 380);            
	FreeNode fn7 = new FreeNode(document, 490, 380);
	double r2 = 20;    

	TwoPointArcOutline twoPointArcOutline = new TwoPointArcOutline(document, fn6, fn7, r2);

	//TwoPointCircleOutline - Линия изображения - окружность, проходящая через два узла с заданным радиусом
	FreeNode fn8 = new FreeNode(document, 440, 420);            
	FreeNode fn9 = new FreeNode(document, 500, 420);
	double r3 = 30;    

	TwoPointCircleOutline twoPointCircleOutline = new TwoPointCircleOutline(document, fn8, fn9, r3);
	twoPointCircleOutline.Color = 200;
	twoPointCircleOutline.PatternName = "WAVES";

	//EllipseArcOutline    - Дуга эллипса        
	FreeNode elNode1 = new FreeNode(document, 580, 320);            
	FreeNode elNode2 = new FreeNode(document, 550, 320);
	FreeNode elNode3 = new FreeNode(document, 610, 320);            
	FreeNode elNode4 = new FreeNode(document, 580, 360);

	EllipseArcOutline ellipseArcOutline = new EllipseArcOutline(document, elNode1, elNode2, elNode3, elNode4);             

	//EllipseOutline - Эллипс по центру и точке
	FreeNode elNode5 = new FreeNode(document, 580, 400);            
	FreeNode elNode6 = new FreeNode(document, 520, 400);
	double elR1 = 25;

	EllipseOutline ellipseOutline = new EllipseOutline(document, elNode5, elNode6, elR1);

	//TwoPointEllipseOutline - Эллипс через две точки и радиус
	FreeNode elNode7 = new FreeNode(document, 560, 460);            
	FreeNode elNode8 = new FreeNode(document, 600, 460);
	double elR2 = 15;

	TwoPointEllipseOutline twoPointEllipseOutline = new TwoPointEllipseOutline(document, elNode7, elNode8, elR2);

	document.EndChanges();//Закрытие блока изменений документа
}
```
