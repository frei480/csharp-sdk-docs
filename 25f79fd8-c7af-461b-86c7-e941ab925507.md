# Узел, построенный по координатам  
  
---  
      
    
    public static void CoordinateNode3D()
    {
        Document document = TFlex.Application.ActiveDocument;//Получение активного документа
        document.BeginChanges("Построение 3D-узла");//Открытие блока изменений документа
    
        //Создаем 3D-узлы
        CoordinateNode3D node1 = new CoordinateNode3D(document); 
        node1.X = 12; node1.Y = 10; node1.Z = 9; 
    
        document.EndChanges();//Закрытие блока изменений документа
    }
