# Вектор привязки  
  
---  
      
    
    public static void FixingVector()//вектор привязки
    {            
        Document document = TFlex.Application.ActiveDocument;//Получение активного документа            
        document.BeginChanges("Построение вектора привязки");//Открытие блока изменений документа
    
        FixingVector fv = new FixingVector(document);
        //создание узлов
        FreeNode fn1 = new FreeNode(document, 50, 400);
        FreeNode fn2 = new FreeNode(document, 200, 400);
    
        fv.StartNode = fn1;//Начальный узел
        fv.EndNode = fn2;//Конечный узел
        fv.Color = 21;//цвет вектора    
    
        document.EndChanges();//Закрытие блока изменений документа
    }
