# Вывод графических объектов  
    
Для вывода графических объектов используются следующие методы:

  * [AddDashLine](M_TFlex_Drawing_Graphics_AddDashLine.md) — добавление в массив штрихов штрихового типа линии сплошного штриха;
  * [AddDashSpace](M_TFlex_Drawing_Graphics_AddDashSpace.md) — добавление в массив штрихов штрихового типа линии пробела (интервала между штрихами);
  * [Arc](Overload_TFlex_Drawing_Graphics_Arc.md) — прорисовка дуги с центром в заданной точке, проходящей через две точки, против часовой стрелки;
  * [Arc](Overload_TFlex_Drawing_Graphics_Arc.md) — прорисовка дуги с центром в заданной точке, проходящей через две точки;
  * [Arc3Points](M_TFlex_Drawing_Graphics_Arc3Points.md) — прорисовка дуги, проходящей через три точки;
  * [Circle](Overload_TFlex_Drawing_Graphics_Circle.md) — прорисовка окружности c использованием центра и радиуса;
  * [Circle](Overload_TFlex_Drawing_Graphics_Circle.md) — прорисовка окружности c использованием центра и точки на окружности;
  * [ClearBackground](M_TFlex_Drawing_Graphics_ClearBackground.md) — очистка фона;
  * [ClearDashes](M_TFlex_Drawing_Graphics_ClearDashes.md) — сброс массива штрихов штрихового типа линии;
  * [DrawMarker](M_TFlex_Drawing_Graphics_DrawMarker.md) — прорисовка маркера в указанной точке;
  * [DrawText](M_TFlex_Drawing_Graphics_DrawText.md) — прорисовка текста в указанной точке;
  * [Fill](M_TFlex_Drawing_Graphics_Fill.md) — заливка области, ограниченной многоугольником;
  * [FillRectangle](M_TFlex_Drawing_Graphics_FillRectangle.md) — заливка прямоугольника;
  * [Line](M_TFlex_Drawing_Graphics_Line.md) — прорисовка отрезка между двумя точками;
  * [LineTo](M_TFlex_Drawing_Graphics_LineTo.md) — прорисовка отрезка от указателя в данную точку;
  * [MoveTo](M_TFlex_Drawing_Graphics_MoveTo.md) — перемещение указателя в данную точку;
  * [Polyline](M_TFlex_Drawing_Graphics_Polyline.md) — прорисовка полилинии;
  * [Rectangle](M_TFlex_Drawing_Graphics_Rectangle.md) — прорисовка границ прямоугольника.



Ниже приведен пример использования методов вывода графических объектов.
    
    
    private void DrawGraphicsObjects(Graphics graphics)
    {
        graphics.BeginDraw();
    
        //Отчищаем фон
        graphics.ClearBackground();
    
        //Устанавливаем зеленый цвет
        graphics.SetColor((int)TFlex.Drawing.Color.Green);
        //Рисуем дугу, проходящую через три заданные точки
        graphics.Arc(new TFlex.Drawing.Point(200.0, 200.0), new TFlex.Drawing.Point(341.0, 200.0), new TFlex.Drawing.Point(59.0, 200.0), false);
    
        //Создаем полилинию
        TFlex.Drawing.Polyline poly = new TFlex.Drawing.Polyline();
        poly.Add(59.0, 200.0);
        poly.Add(100.0, 220.0);
        poly.Add(300.0, 220.0);
        poly.Add(341.0, 200.0);
        //Прорисовываем полилинию
        graphics.Polyline(poly);
        //Закращиваем область внутри полилинии зеленым цветом
        graphics.Fill(poly);
    
        //Создаем прямоугольник
        TFlex.Drawing.Rectangle rect = new TFlex.Drawing.Rectangle(100.0, 100.0, 200.0, 200.0);
        //Устанавливаем голубой цвет
        graphics.SetColor((int)TFlex.Drawing.Color.LightBlue);
        //Заполняем прямоугольник голубым цветом
        graphics.FillRectangle(rect);
        //Устанавливаем красный цвет
        graphics.SetColor((int)TFlex.Drawing.Color.Red);
        //Устанавливаем толщину линии рисования
        graphics.SetLineWidth(3.0);
        //Устанавливаем прерывистую линию
        graphics.DashedType = true;
        //Длина штриха прерывистой линии
        graphics.AddDashLine(10.0);
        //Длина пробела прерывистой линии
        graphics.AddDashSpace(5.0);
        //Для выбранного цвета и типа линии создаем рамку вокруг голубого прямоугольника
        graphics.Rectangle(rect);
        //Снова устанавливаем сплошную линию
        graphics.ClearDashes();
    
        //Задаем цвет в RGB формате
        graphics.SetRGBColor(255, 0, 255);
        //Рисуем окружность по центру и точке на ней
        graphics.Circle(new TFlex.Drawing.Point(250.0, 150.0), new TFlex.Drawing.Point(250.0, 100.0));
    
        //В центре окружности ставим крестик
        graphics.DrawMarker(TFlex.Drawing.MarkerType.NodeCross, new TFlex.Drawing.Point(250.0, 150.0));
    
        graphics.EndDraw();
    }
