# Удаление объектов  
  
---  
  
Для удаления объектов документа можно воспользоваться методом [DeleteObjects](M_TFlex_Model_Document_DeleteObjects.md). Для его вызова необходимо заполнить массив удаляемых объектов и сконструировать объект, задающий опции удаления объектов.

Ниже приведен пример кода функции, которая удаляет все линии изображения на текущей странице документа без вопроса об удалении зависимых от этих линий изображения объектов.
    
    
    private void DeleteAllOutlines(Document document)
    {
        //Создаем массив удаляемых объектов
        ObjectArray DelArray = new ObjectArray();
        //Перебор всех линий изображения в документе
        foreach (Object2D outline in document.GetOutlines())
        {
            //Если найденная линия находится на активной странице,
            if (outline.Page == document.ActivePage)
                //то заносим её в массив удаляемых объектов
                DelArray.Add(outline);
        }
        //Удаляем выбранные линии без вопроса об удалении зависимых от этих линий изображения объектов
        document.DeleteObjects(DelArray, new DeleteOptions(true));
    }
    
    
    Private Sub DeleteAllOutlines(ByVal document As Document)
        'Создаем массив удаляемых объектов
        Dim DelArray As ObjectArray
        DelArray = New ObjectArray()
        'Перебор всех линий изображения в документе
        For Each outline As Object2D In document.GetOutlines()
            'Если найденная линия находится на активной странице,
            If outline.Page Is document.ActivePage Then
                'то заносим её в массив удаляемых объектов
                DelArray.Add(outline)
            End If
        Next
        'Удаляем выбранные линии без вопроса об удалении зависимых от этих линий изображения объектов
        document.DeleteObjects(DelArray, New DeleteOptions(True))
    End Sub
